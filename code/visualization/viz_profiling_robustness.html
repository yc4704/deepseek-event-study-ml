from scipy.stats import ttest_ind, mannwhitneyu

numeric_cols = data.select_dtypes(include=[np.number]).columns.tolist()

test_results = []

for col in numeric_cols:
    v = data[data['label'] == 'Vulnerable'][col].dropna()
    r = data[data['label'] == 'Resilient'][col].dropna()
    
    if len(v) > 5 and len(r) > 5:
        t_stat, t_p = ttest_ind(v, r, equal_var=False)
        u_stat, u_p = mannwhitneyu(v, r, alternative='two-sided')
        test_results.append([col, v.mean(), r.mean(), r.mean() - v.mean(), t_p, u_p])

test_results = pd.DataFrame(test_results, columns=[
    'feature', 'vulnerable_mean', 'resilient_mean', 'difference', 't_test_p', 'mannwhitney_p'
])
